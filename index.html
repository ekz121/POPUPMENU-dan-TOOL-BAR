    private void kirimActionPerformed(java.awt.event.ActionEvent evt) {                                      
       // Mengambil teks dari masing-masing JTextField
    String nim = jTextField1.getText();
    String nama = jTextField2.getText();
    String keterangan = jTextField3.getText();

    // Validasi sederhana: Jika NIM atau NAMA kosong
    if (nim.isEmpty() || nama.isEmpty()) {
        JOptionPane.showMessageDialog(this, "NIM dan Nama harus diisi!", "Error", JOptionPane.ERROR_MESSAGE);
    } else {
        // Menampilkan notifikasi sukses dengan ringkasan data
        String pesan = "Data berhasil dikirim!\n\n" +
                       "NIM: " + nim + "\n" +
                       "Nama: " + nama + "\n" +
                       "Keterangan: " + keterangan;
        
        JOptionPane.showMessageDialog(this, pesan, "Konfirmasi Pengiriman", JOptionPane.INFORMATION_MESSAGE);
    }

    }
